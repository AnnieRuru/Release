function	script	getitemname2	{
	freeloop true;
	deletearray @itemname2_info$;
	.@id = getarg(0);
	.@type = getiteminfo( .@id, ITEMINFO_TYPE );
	.@refine = getarg(2);
	.@card[0] = getarg(4);
	.@card[1] = getarg(5);
	.@card[2] = getarg(6);
	.@card[3] = getarg(7);
	.@bound = getarg(8,0);
	.@ref$ = ( .@refine )? "+"+ .@refine +" " : "" ; // I think no need refinement check for IT_WEAPON or IT_ARMOR, the source automatically convert refined etc items into no refinement
	.@slots$ = ( .@slot = getitemslots( .@id ) )? " ["+ .@slot +"]" : "" ;

	if ( .@bound ) {
		@itemname2_info$[0] = "^CCAA00"; // this condition always get overwritten ...
		if ( .@bound == 1 )
			@itemname2_info$[5] = "Account";
		else if ( .@bound == 2 )
			@itemname2_info$[5] = "Guild";
		else if ( .@bound == 3 )
			@itemname2_info$[5] = "Party";
		else if ( .@bound == 4 )
			@itemname2_info$[5] = "Character";
	}

	if ( .@type != IT_WEAPON && .@type != IT_ARMOR && .@type != IT_PETEGG && .@card[0] != 254 ) // non-equipable item that isn't signed
		.@return$ = getitemname( .@id );
	else if ( ( .@type == IT_WEAPON || .@type == IT_ARMOR ) && !.@card[0] && !.@card[1] && !.@card[2] && !.@card[3] ) // equipment with no card, for reduce gotocount
		.@return$ = .@ref$ + getitemname( .@id ) + .@slots$;
	else if ( ( .@type == IT_WEAPON || .@type == IT_ARMOR ) && ( .@card[0] || .@card[1] || .@card[2] || .@card[3] ) && .@card[0] != 254 && .@card[0] != 255 ) { // equipment that has cards, but not a signed item
		for ( .@i = 0; .@i < 4; ++.@i ) {
			if ( !.@cnt[ .@card[.@i] ] )
				.@cardlist[.@cardlistsize++] = .@card[.@i];
			++.@cnt[ .@card[.@i] ];
			.@enchant += ( .@card[.@i] && .@i >= .@slot && getstrlen( $@cardprefix$[.@card[.@i]] ) );
		}
		for ( .@i = 0; .@i < .@cardlistsize; ++.@i ) {
			if ( .@cnt[ .@cardlist[.@i] ] == 2 )
				.@cnt$[.@i] = "Double";
			else if ( .@cnt[ .@cardlist[.@i] ] == 3 )
				.@cnt$[.@i] = "Triple";
			else if ( .@cnt[ .@cardlist[.@i] ] == 4 )
				.@cnt$[.@i] = "Quadruple";
			if ( !compare( $@cardprefix$[ .@cardlist[.@i] ], "of " ) )
				.@prefix$ = .@prefix$ + .@cnt$[.@i] +( ( getstrlen(.@cnt$[.@i]) )? " ": "" )+ $@cardprefix$[ .@cardlist[.@i] ] +" ";
			else
				.@postfix$ = .@postfix$ +" "+ $@cardprefix$[ .@cardlist[.@i] ] +( ( getstrlen(.@cnt$[.@i]) )? " ": "" )+ .@cnt$[.@i];
		}
		if ( .@enchant == 1 )
			@itemname2_info$[0] = "^66CC00";
		else if ( .@enchant == 2 )
			@itemname2_info$[0] = "^0066CC";
		else if ( .@enchant == 3 )
			@itemname2_info$[0] = "^CC6600";
		else if ( .@enchant == 4 )
			@itemname2_info$[0] = "^CC44CC";
		.@return$ = .@ref$ + .@prefix$ + getitemname( .@id ) + .@postfix$ + .@slots$;
	}
	else if ( .@card[0] == 254 || .@card[0] == 255 ) { // signed items
		.@cid = @itemname2_info$[1] = .@card[2];
		.@aid = charid2rid(.@cid);
		if ( .@aid )
			.@name$ = rid2name(.@aid);
		else if ( !query_sql("select name from `char` where char_id = "+ .@cid, .@name$) )
			.@name$ = "Unknown";
		.@sc = .@card[1] / 10;
		if ( .@sc == 128 ) {
			.@sc$ = "Very Strong ";
			@itemname2_info$[3] = "1";
		}
		else if ( .@sc == 256 ) {
			.@sc$ = "Very Very Strong ";
			@itemname2_info$[3] = "2";
		}
		else if ( .@sc == 384 ) {
			.@sc$ = "Very Very Very Strong ";
			@itemname2_info$[3] = "3";
		}
		.@ele = .@card[1] % 10;
		if ( .@ele == 1 )
			.@ele$ = @itemname2_info$[4] = "Ice ";
		else if ( .@ele == 2 )
			.@ele$ = @itemname2_info$[4] = "Earth ";
		else if ( .@ele == 3 )
			.@ele$ = @itemname2_info$[4] = "Fire ";
		else if ( .@ele == 4 )
			.@ele$ = @itemname2_info$[4] = "Wind ";
		.@return$ = .@ref$ + .@sc$ + .@name$ +"'s "+ .@ele$ + getitemname( .@id );
	}
	else if ( .@type == IT_PETEGG ) {
		@itemname2_info$[1] = .@card[1] + .@card[2] * (1<<16);
		.@return$ = getitemname( .@id );
	}

	if ( !getarg(1) ) { // unidentified
		if ( .@type == IT_ARMOR ) {
			@itemname2_info$[0] = "^999999";
			@itemname2_info$[2] = .@return$;
			.@loc = getiteminfo( .@id, ITEMINFO_LOC );
			if ( getstrlen( $@unidentified$[.@id] ) )
				.@return$ = $@unidentified$[.@id];
			else if ( .@loc & (EQP_HEAD_LOW|EQP_HEAD_MID|EQP_HEAD_TOP) )
				.@return$ = "Headgear";
			else if ( .@loc & EQP_ARMOR )
				.@return$ = "Armor";
			else if ( .@loc & EQP_SHIELD )
				.@return$ = "Shield";
			else if ( .@loc & EQP_GARMENT )
				.@return$ = "Garment";
			else if ( .@loc & EQP_SHOES )
				.@return$ = "Shoes";
			else if ( .@loc & (EQP_ACC_L|EQP_ACC_R) )
				.@return$ = "Accessory";
			else
				.@return$ = "Equipment - ERROR";
		}
		else if ( .@type == IT_WEAPON ) {
			@itemname2_info$[0] = "^999999";
			@itemname2_info$[2] = .@return$;
			if ( .@view == W_DAGGER )
				.@return$ = "Dagger";
			else if ( .@view == W_1HSWORD )
				.@return$ = "Sword";
			else if ( .@view == W_2HSWORD )
				.@return$ = "Two-handed Sword";
			else if ( .@view == W_1HSPEAR || .@view == W_2HSPEAR )
				.@return$ = "Spear";
			else if ( .@view == W_1HAXE || .@view == W_2HAXE )
				.@return$ = "Axe";
			else if ( .@view == W_MACE )
				.@return$ = "Mace";
			else if ( .@view == W_STAFF || .@view == W_2HSTAFF )
				.@return$ = "Staff";
			else if ( .@view == W_BOW )
				.@return$ = "Bow";
			else if ( .@view == W_KNUCKLE )
				.@return$ = "Knuckle";
			else if ( .@view == W_MUSICAL )
				.@return$ = "Instrument";
			else if ( .@view == W_WHIP )
				.@return$ = "Whip";
			else if ( .@view == W_BOOK )
				.@return$ = "Book";
			else if ( .@view == W_KATAR )
				.@return$ = "Katar";
			else if ( .@view == W_REVOLVER )
				.@return$ = "Revolver";
			else if ( .@view == W_RIFLE )
				.@return$ = "Rifle";
			else if ( .@view == W_GATLING )
				.@return$ = "Gatling Gun";
			else if ( .@view == W_SHOTGUN )
				.@return$ = "Shotgun";
			else if ( .@view == W_GRENADE )
				.@return$ = "Grenade Launcher";
			else if ( .@view == W_HUUMA )
				.@return$ = "Huuma Shuriken";
			else
				.@return$ = "Weapon - ERROR";
		}
	}
//	else if ( getarg(3) ) // broken equipment ... seems no longer show red
//		@itemname2_info$[0] = "^FF0000";

	if ( !getstrlen( @itemname2_info$[0] ) )
		@itemname2_info$[0] = "^000000";

	freeloop false;

	return .@return$;
}
